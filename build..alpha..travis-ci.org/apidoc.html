    templatizer = function (input, output, options, done) {
    var args = arguments;

    if (args.length === 3) {
        // input, output, done
        if (_.isString(args[1]) && _.isFunction(args[2])) {
            done = args[2];
            options = null;
        }
        // input, options, done
        else if (_.isObject(args[1]) && _.isFunction(args[2])) {
            done = args[2];
            options = args[1];
            output = null;
        }
    } else if (arguments.length === 2) {
        // input, done
        if (_.isFunction(args[1])) {
            done = args[1];
            options = null;
            output = null;
        }
        // input, options
        else if (_.isObject(args[1])) {
            options = args[1];
            done = null;
            output = null;
        }
    }

    // Default values for done and options so we dont error
    done || (done = function (err, compiledOutput) {
        if (err) {
            throw err;
        } else {
            process.stdout.write(compiledOutput);
        }
    });
    options || (options = {});
